<script>
    window.onload = function(){
        var nav = document.getElementById('nav');
        var lis = document.querySelectorAll('li');
        var clicked = true;

        lis[lis.length-1].onclick = function(){
            for(var i = 0; i < lis.length;i ++) {
                var n = i - lis.length / 2;
                var deg;
                if(clicked){
                    deg = n * 15;
                }else{
                    deg = 360;
                }
                lis[i].style.transform = 'rotate('+deg+'deg)';
            }
            clicked = !clicked;
        };

        for(var i = 0;i < lis.length - 1; i++){
            lis[i].index = i;
            lis[i].onclick = function(){
                var leftDeg = 0;
                var rightDeg = 15;
                this.style.transform = 'rotate(0deg)';
                for(var i = this.index - 1;i >= 0;i--){
                    leftDeg = leftDeg-15;
                    lis[i].style.transform = 'rotate('+leftDeg+'deg)';
                }
                for(var j = this.index + 1;j < lis.length;j ++){
                    rightDeg = rightDeg+15;
                    lis[j].style.transform = 'rotate('+rightDeg+'deg)';
                }
            }
        }

        var cav = document.getElementById('canvas');
        cav.width = window.innerWidth;
        cav.height = window.innerHeight + 400;
        var context = canvas.getContext('2d');
        var balls = [];
        var timer;
        var colors = ['#69d2e7','#a7dbdb','#e0e4cc','#f38630','#fa6900','#ff4e50','#f9d423'];
        /*
        *var ball = {
        * x:x的位置
        * y:y的位置
        * r:圆的半径
        * vx:x轴的速度
        * vy:y轴的速度
        * color:圆的颜色
        * }
        * 角度转弧度：角度*π/180
         */
        function draw(ball){
            context.beginPath();
            //arc(x轴位置,y轴位置,r,起始弧度，结束弧度)
            context.arc(ball.x,ball.y,ball.r,0,Math.PI*2);
            context.fillStyle = ball.color;
            context.globalCompositeOperation = 'lighter';//合成
            context.fill();
        }
        //取到x到y之间随机数：Math.random()*(x-y)+x)
        function random(min,max){
            return Math.round(Math.random()*(max-min)+min);
        }
        var on = true; //鼠标移动的时候
        canvas.onmousemove = function(e){
            for(var i = 0; i < 2;i ++){
                var ball = {
                    x: random(-5,5)+ e.clientX,
                    y: random(-5,5) + e.clientY + window.pageYOffset,//pageYOffset
                    r: random(10,45),
                    vx: Math.random()-0.5,
                    vy: Math.random()-0.5,
                    color: colors[random(0,colors.length-1)]
                };
                balls.push(ball);
                if(balls.length > 300){
                    balls.shift();//shift()技巧?
                }
            }
            if(on){ //让定时器开启一次
                clearInterval(timer);
                timer = setInterval(drawball,30);
                on = false;
            }

            function drawball(){
                context.clearRect(0,0,canvas.width,canvas.height);
                for(var i = 0;i < balls.length;i ++){
                    balls[i].x += balls[i].vx*8;    //通过速度改变x轴的位置
                    balls[i].y += balls[i].vy*8;
                    balls[i].r = balls[i].r*0.94;   //改变球的半径

                    balls[i].index = i;  //如果小球的半径小于1，不让canvas再画他
                    //添加索引方便下面找到他并将它删除
                    if(balls[i].r < 1){
                        balls.splice(balls[i].index,1);
                        continue;//如果小球已经被删了，以下的代码不执行
                    }
                    draw(balls[i]);
                    //如果balls没有东西了，就清除定时器
                    if(!balls.length){
                        clearInterval(timer);
                        on = true;//让on的值与定时器保持一致
                    }
                }
            }
        }
    }
</script>
