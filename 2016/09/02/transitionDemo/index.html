<script>
    window.onload = function(){
        var btn = document.getElementById('run');
        var imgs = document.querySelectorAll('img');
        var on = true;//true代表可以点，false代表不可点
        btn.onclick = function(){
            if(!on){
                return;
            }
            on = false;
            var endNum = 0;
            var allEnd = 0;
            for(var i = 0;i < imgs.length;i ++){
                (function(i){
                    setTimeout(function(){
                        motion(imgs[i],'100ms',function(){
                            this.style.transform = 'scale(0)';
                        },function(){
                            motion(this,'1s',function(){
                                this.style.transform = 'scale(1)';
                                this.style.opacity = 0;
                                endNum ++;
                                if(endNum === imgs.length){
                                    toBig();
                                }
                            });
                        })
                    },Math.random()*1000);
                })(i);
            }
            function toBig(){
                for(var i = 0; i < imgs.length; i ++){
                    imgs[i].style.transition = '';
                    //想要一个物体有css3中的效果，要给他设置初始值
                    imgs[i].style.transform = 'rotateY(0deg) translateZ(-'+Math.random()*500+'px)';

                    (function(i){
                        setTimeout(function(){
                            motion(imgs[i],'2s',function(){
                                this.style.opacity = 1;
                                this.style. transform = 'rotateY(-360deg) translateZ(0)';
                            },function(){
                                allEnd ++;
                                if(allEnd === imgs.length){
                                    on = true;//所有运动完了之后可以点
                                }
                            })
                        },Math.random()*1000)
                    })(i)
                }
            }
        };
        function motion(obj,timer,doFn,cb){
            obj.style.transition = timer;
            doFn.call(obj);
            var called = false;
            obj.addEventListener('transitionend',function(){
                if(!called){
                    cb && cb.call(obj);
                    called = true;
                }
            },false); //false:事件冒泡与捕获
        }
    };
</script>
