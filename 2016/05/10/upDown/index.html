<script  type="text/javascript">
  $(function(){
    var upDown = {
      init: function(){
        this.bindEvent();
        this.reline();
      },
      reline: function(){
        var edits = $('.edit'),
          i = 0;
          l = edits.length;
        for(;i < l; i++){
          $('.up').show().next('span').show();
          $('.down').show().next('span').show();
          $(edits[0]).find('.up').hide().next('span').hide();
          $(edits[l-1]).find('.down').hide().next('span').hide();
        }
      },
      bindEvent: function(){

        $('.row').on('click','.up',function(){
          var that = $(this);
          var _new = that.closest('.row');
          var _old = that.closest('.row').prev('.row');
          if (_new.length > 0) {
            var _temp = _old.html();
            _old.empty().append(_new.html());
            _new.empty().append(_temp);
          }
          upDown.reline();
        });

        $('.row').on('click','.down',function(){
          var that = $(this);
          var _new = that.closest('.row');
          var _old = that.closest('.row').next('.row');
          if (_new.length > 0) {
            var _temp = _old.html();
            _old.empty().append(_new.html());
            _new.empty().append(_temp);
          }
          upDown.reline();
        });

        $('.row').on('click','.del',function(){
          var that = $(this);
          that.closest('.row').remove();
        });
      }
    }
    upDown.init();
  })
</script>
