                <li class="records">我喜欢站在未完工的两广路上喊你的名字</li>
                <li class="records" id="nice" data-words="美好">今天是美好的一天。</li>
                <li class="records" id="happy" data-words="快乐">希望你每天都快乐。</li>

<script type="text/javascript" src="jquery-1.7.2.js"></script>
<script>
    $(document).ready(function(){
        var palyer = {
            init: function(){
                this.sum = 0;
                this.records = $('.records');
                this.n = $('.records').length;
                this.words = $('.words');
                this.matchEvent();
                this.playerInit();
                this.trunRed();
            },
            trunRed: function(){
                for(var i = 0;i < this.n; i ++){
                    var words = $(this.records[i]).attr('data-words');
                    var text = $(this.records[i]);
                    if(words){
                        text.html(this.eachColor(text,words));
                    }
                }
            },
            eachColor: function(p,t){
                var nt = '<span style="color:red">'+t+"</span>";
                var reg = RegExp(t,"g");
                return  p.text().replace(reg,nt);
            },
            playerInit: function(){
                var that = this;
                that.timer = setInterval(function(){
                    $('.timing').text(that.sum);
                    that.sum ++;
                   if(that.sum === 31){
                       clearInterval(that.timer);
                   }
                    that.recordsMatch(that.sum);
                },1000);
            },
            recordsMatch: function(sum){
                for(var i = 0;i < this.n; i ++){
                    this.records.removeClass('active');
                    if(0 <= sum && sum <= 5){
                        $(this.records[0]).addClass('active');
                    }
                    if(5 < sum && sum <= 10){
                        $(this.records[1]).addClass('active');
                    }
                    if(10 < sum && sum <= 15){
                        $(this.records[2]).addClass('active');
                    }
                    if(15 < sum && sum <= 20){
                        $(this.records[3]).addClass('active');
                    }
                    if(20 < sum && sum <= 30){
                        $(this.records[4]).addClass('active');
                    }
                }
            },
            jumpTo: function(keys){
                for(var i = 0;i < this.n; i ++){
                    var id = $(this.records[i]).attr('id');
                    this.records.removeClass('active');
                    if(id === keys){
                        $(this.records[i]).addClass('active');
                        return;
                    }
                }
            },
            matchEvent: function(){
                var that = this;
                this.words.each(function(e){
                    $(this).on('click',function(){
                        var keys = $(this).attr('data-key');
                        that.jumpTo(keys);
                    })
                })

            }
        }
        palyer.init();
    })
</script>
